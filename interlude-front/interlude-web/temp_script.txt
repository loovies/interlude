  // жҢүжё…жҷ°еәҰжҺ’еәҸпјҲд»Һй«ҳеҲ°дҪҺпјү
  const order: Record<string, number> = {
    '4k': 1,
    '2160p': 2,
    '1440p': 3,
    '1080p': 4,
    '720p': 5,
    '480p': 6,
    '360p': 7,
    '240p': 8,
  }
  return qualities.sort((a, b) => {
    const orderA = order[a.quality.toLowerCase()] || 999
    const orderB = order[b.quality.toLowerCase()] || 999
    return orderA - orderB
  })
})

// еҲқе§ӢеҢ–ж’ӯж”ҫеҷЁ

  console.log('еҲқе§ӢеҢ–жҠ–йҹійЈҺж јж’ӯж”ҫеҷЁ')

      player.value.pause()

      height: '100%',

      playbackRate: [0.5, 0.75, 1.0, 1.25, 1.5, 2.0],
      lang: 'zh-cn',

      // ж·»еҠ еҠ иҪҪе’Ңй”ҷиҜҜеӨ„зҗ?

      startTime: 0,
      // зҪ‘з»ңдјҳеҢ–
      maxBufferLength: 30, // жңҖеӨ§зј“еҶІй•ҝеә?0з§?
      minBufferLength: 5,  // жңҖе°Ҹзј“еҶІй•ҝеә?з§?
      bufferBehind: 10,    // еҗ‘еҗҺзј“еҶІ10з§?
      // ж·»еҠ е°ұз»Әеӣһи°ғ - еҸӘи®ҫзҪ®дёҖж¬ЎпјҢйҒҝе…ҚйҮҚеӨҚи®ҫзҪ®
      ready: () => {
        console.log('XGPlayerж’ӯж”ҫеҷЁе°ұз»?)
        // иҝҷйҮҢдёҚи®ҫзҪ?loading.value = falseпјҢз”ұ canplay дәӢд»¶еӨ„зҗҶ
      },
      // ж·»еҠ й”ҷиҜҜеӨ„зҗҶ
      error: (err: any) => {
        console.error('XGPlayerж’ӯж”ҫй”ҷиҜҜ:', err)

        if (currentBackupIndex.value < backupVideoSources.length - 1) {

        } else {
          error.value = 'и§Ҷйў‘еҠ иҪҪеӨұиҙҘпјҢиҜ·жЈҖжҹҘзҪ‘з»ңиҝһжҺ?
        }
      }
    })

    if (player.value) {
      // е…Ҳз§»йҷӨеҸҜиғҪеӯҳеңЁзҡ„ж—§дәӢд»¶зӣ‘еҗ¬еҷЁ
      // XGPlayerзҡ„offж–№жі•йңҖиҰҒдәӢд»¶зұ»еһӢе’Ңеӣһи°ғеҮҪж•°пјҢиҝҷйҮҢжҲ‘д»¬зӣҙжҺҘйҮҚж–°з»‘е®?

        if (now - lastBufferingTime.value < 10000) {
          if (bufferingCount.value >= 3) {
            networkQuality.value = 'poor'
            console.log('зҪ‘з»ңиҙЁйҮҸиҫғе·®пјҢиҖғиҷ‘йҷҚзә§жё…жҷ°еә?)
            
            // еҰӮжһңеҗҜз”ЁиҮӘеҠЁйҷҚзә§дё”еҪ“еүҚдёҚжҳҜжңҖдҪҺжё…жҷ°еәҰ
            if (autoDegradeEnabled.value && sortedQualities.value.length > 1) {
              const currentIndex = sortedQualities.value.findIndex(q => q.url === getCurrentVideoUrl())
              if (currentIndex > 0) { // дёҚжҳҜжңҖдҪҺжё…жҷ°еәҰ

                  degradeQuality()

              }

          } else if (bufferingCount.value >= 2) {
            networkQuality.value = 'medium'
          }
        } else {
          // йҮҚзҪ®и®Ўж•°
          bufferingCount.value = 1
        }
        
        lastBufferingTime.value = now
      })

      player.value.on('playing', () => {
        console.log('и§Ҷйў‘ејҖе§Ӣж’ӯж”?)
        loading.value = false
        
        // зӣ‘жҺ§зҪ‘з»ңжҒўеӨҚ
        if (networkQuality.value === 'poor') {
          // еҰӮжһңиҝһз»ӯж’ӯж”ҫ30з§’жІЎжңүзј“еҶІпјҢзҪ‘з»ңеҸҜиғҪжҒўеӨҚдә?

              bufferingCount.value = 0

      const loadingTimeout = setTimeout(() => {

          console.log('и§Ҷйў‘еҠ иҪҪи¶…ж—¶')
          loading.value = false
          error.value = 'и§Ҷйў‘еҠ иҪҪи¶…ж—¶пјҢиҜ·жЈҖжҹҘзҪ‘з»ңжҲ–и§Ҷйў‘ең°еқҖ'

      }, 5000) // еҮҸе°‘еҲ?з§’и¶…ж—¶пјҢжӣҙеҝ«еҸҚйҰҲ

  } catch (err: any) {

// еӨҮз”Ёи§Ҷйў‘жәҗеҲ—иЎЁпјҲеҪ“дё»жәҗеҠ иҪҪеӨұиҙҘж—¶дҪҝз”Ёпј? дҪҝз”Ёжӣҙеҝ«зҡ„и§Ҷйў‘жәҗ
const backupVideoSources = [

  'https://storage.googleapis.com/gtv-videos-bucket/sample/ForBiggerFun.mp4', // иҫғзҹӯи§Ҷйў‘пјҢеҝ«йҖҹеҠ иҪ?
  'https://storage.googleapis.com/gtv-videos-bucket/sample/ForBiggerBlazes.mp4', // иҫғзҹӯи§Ҷйў‘
  'https://storage.googleapis.com/gtv-videos-bucket/sample/ForBiggerEscapes.mp4', // иҫғзҹӯи§Ҷйў‘

  'https://storage.googleapis.com/gtv-videos-bucket/sample/ElephantsDream.mp4', // й•ҝи§Ҷйў‘пјҢж”ҫеңЁеҗҺйқў
]

// еҪ“еүҚдҪҝз”Ёзҡ„еӨҮз”Ёжәҗзҙўеј•
const currentBackupIndex = ref(0)

// иҺ·еҸ–еҪ“еүҚи§Ҷйў‘URL
const getCurrentVideoUrl = () => {
  if (sortedQualities.value.length === 0) {
    console.warn('жІЎжңүеҸҜз”Ёзҡ„жё…жҷ°еәҰж•°жҚ®пјҢдҪҝз”ЁеӨҮз”Ёжәҗ')
    // иҝ”еӣһеӨҮз”Ёи§Ҷйў‘URL
    return backupVideoSources[currentBackupIndex.value % backupVideoSources.length]
  }

  const qualityData = sortedQualities.value.find((q) => q.quality === currentQuality.value)
  if (!qualityData) {
    console.warn('жңӘжүҫеҲ°еҪ“еүҚжё…жҷ°еәҰпјҢдҪҝз”Ёй»ҳи®Өжё…жҷ°еәҰ:', sortedQualities.value[0].quality)
    const defaultUrl = sortedQualities.value[0].url || backupVideoSources[currentBackupIndex.value % backupVideoSources.length]
    return defaultUrl
  }

  const url = qualityData.url || backupVideoSources[currentBackupIndex.value % backupVideoSources.length]
  console.log('еҪ“еүҚи§Ҷйў‘URL:', url)
  return url
}

// еҲҮжҚўеҲ°дёӢдёҖдёӘеӨҮз”Ёжәҗ
const switchToNextBackupSource = () => {
  currentBackupIndex.value = (currentBackupIndex.value + 1) % backupVideoSources.length
  console.log('еҲҮжҚўеҲ°еӨҮз”Ёи§Ҷйў‘жәҗ:', backupVideoSources[currentBackupIndex.value])
  
  // йҮҚж–°еҲқе§ӢеҢ–ж’ӯж”ҫеҷЁ
  if (player.value) {
    player.value.destroy()
    player.value = null
    setTimeout(() => {
      initPlayer()
    }, 500)
  }
}

// и®ҫзҪ®еҲқе§Ӣжё…жҷ°еә?
const setInitialQuality = () => {
  if (sortedQualities.value.length > 0) {
    // й»ҳи®ӨйҖүжӢ©жңҖй«ҳжё…жҷ°еәҰ
    const defaultQuality = sortedQualities.value[0].quality
    currentQuality.value = defaultQuality
  }
}

// з”ҹжҲҗеј№е№•ж•°жҚ®
const generateDanmuData = () => {
  danmuList.value = generateMockDanmu(props.videoData.videoId, totalDuration.value)
}

// и®ҫзҪ®ж’ӯж”ҫеҷЁдәӢд»?
const setupPlayerEvents = () => {
  if (!player.value) return

  player.value.on('play', () => {
    isPlaying.value = true
    emit('play')
  })

  player.value.on('pause', () => {
    isPlaying.value = false
    emit('pause')
  })

  player.value.on('ended', () => {
    isPlaying.value = false
    emit('ended')
  })

  player.value.on('error', (err: any) => {
    console.error('и§Ҷйў‘ж’ӯж”ҫй”ҷиҜҜ:', err)
    error.value = getErrorMessage(err)
    emit('error', err)
  })

  player.value.on('timeupdate', () => {
    if (player.value) {
      currentTime.value = player.value.currentTime
      totalDuration.value = player.value.duration || props.videoData.duration || 0

      if (totalDuration.value > 0) {
        progressPercent.value = (currentTime.value / totalDuration.value) * 100
      }
    }
  })

  player.value.on('volumechange', () => {
    if (player.value) {
      volume.value = player.value.volume
      isMuted.value = player.value.muted
    }
  })

  player.value.on('fullscreenchange', (isFull: boolean) => {
    isFullscreen.value = isFull
  })

  // зј“еҶІзӣ‘жҺ§
  player.value.on('progress', () => {
    if (player.value) {
      const buffered = player.value.buffered
      if (buffered && buffered.length > 0) {
        const bufferedEnd = buffered.end(buffered.length - 1)
        const bufferedPercent = (bufferedEnd / totalDuration.value) * 100
        
        // еҸӘеңЁи°ғиҜ•ж—¶жҳҫзӨ?
        if (bufferedPercent % 10 < 1) { // жҜ?0%жҳҫзӨәдёҖж¬?
          console.log(`зј“еҶІиҝӣеәҰ: ${bufferedPercent.toFixed(1)}% (${bufferedEnd.toFixed(1)}/${totalDuration.value})`)
        }
        
        // жҷәиғҪйў„еҠ иҪҪзӯ–з•?
        const bufferAhead = bufferedEnd - currentTime.value
        
        if (bufferAhead < 5) { // зј“еҶІдёҚи¶і5з§?
          console.log('зј“еҶІдёҚи¶іпјҢеү©дҪҷзј“еҶ?', bufferAhead.toFixed(1), 'з§?)
          
          // еҰӮжһңзҪ‘з»ңиҙЁйҮҸеҘҪпјҢе°қиҜ•йў„еҠ иҪҪжӣҙеӨ?
          if (networkQuality.value === 'good' && player.value.video) {
            // е°қиҜ•йў„еҠ иҪҪдёӢдёҖж®?
            try {
              // XGPlayer еҸҜиғҪдёҚж”ҜжҢҒзӣҙжҺҘжҺ§еҲ¶йў„еҠ иҪҪпјҢиҝҷйҮҢеҸӘжҳҜи®°еҪ?
              console.log('е°қиҜ•йў„еҠ иҪҪжӣҙеӨҡж•°жҚ?..')
            } catch (e) {
              console.log('йў„еҠ иҪҪеӨұиҙ?', e)
            }
          }
        }
      }
    }
  })

  // зӣ‘еҗ¬й”®зӣҳдәӢд»¶
  document.addEventListener('keydown', handleKeydown)
}

// жҳҫзӨәжҺ§еҲ¶жқ?
const showControls = () => {
  controlsVisible.value = true
  if (hideTimer.value) {
    clearTimeout(hideTimer.value)
  }

  // еҰӮжһңдёҚжҳҜж’ӯж”ҫзҠ¶жҖҒпјҢдҝқжҢҒжҳҫзӨә
  if (isPlaying.value) {
    hideTimer.value = setTimeout(() => {
      controlsVisible.value = false
    }, 5000) // 5з§’еҗҺйҡҗи—Ҹ
  }
}

// ејҖе§Ӣйҡҗи—Ҹи®Ўж—¶еҷЁ
const startHideTimer = () => {
  if (hideTimer.value) {
    clearTimeout(hideTimer.value)
  }

  if (isPlaying.value) {
    hideTimer.value = setTimeout(() => {
      controlsVisible.value = false
    }, 5000) // 5з§’еҗҺйҡҗи—Ҹ
  }
}

// зӮ№еҮ»иҝӣеәҰжқЎи·іиҪ?
const seekToPosition = (event: MouseEvent) => {
  if (!player.value || !progressBar.value) return

  const rect = progressBar.value.getBoundingClientRect()
  const clickX = event.clientX - rect.left
  const width = rect.width
  const percent = clickX / width

  if (totalDuration.value > 0) {
    const seekTime = totalDuration.value * percent
    player.value.currentTime = seekTime
    currentTime.value = seekTime
  }
}

const changeVolume = (event: MouseEvent) => {
  if (!player.value || !volumeSlider.value) return

  const rect = volumeSlider.value.getBoundingClientRect()
  const clickX = event.clientX - rect.left
  const width = rect.width
  let newVolume = clickX / width

  // йҷҗеҲ¶йҹійҮҸең?-1д№Ӣй—ҙ
  newVolume = Math.max(0, Math.min(1, newVolume))

  volume.value = newVolume
  player.value.volume = newVolume
  player.value.muted = newVolume === 0
}

// ж”№еҸҳйҹійҮҸпјҲз«–зҡ„пјү
const changeVolumeVertical = (event: MouseEvent) => {
  if (!player.value || !volumeSlider.value) return

  const rect = volumeSlider.value.getBoundingClientRect()
  const clickY = event.clientY - rect.top
  const height = rect.height
  let newVolume = 1 - (clickY / height) // д»ҺдёӢеҫҖдёҠеўһеҠ?

  // йҷҗеҲ¶йҹійҮҸең?-1д№Ӣй—ҙ
  newVolume = Math.max(0, Math.min(1, newVolume))

  volume.value = newVolume
  player.value.volume = newVolume
  player.value.muted = newVolume === 0
}

// ж јејҸеҢ–ж—¶й—?
const formatTime = (seconds: number) => {
  if (!seconds && seconds !== 0) return '00:00'

  const hrs = Math.floor(seconds / 3600)
  const mins = Math.floor((seconds % 3600) / 60)
  const secs = Math.floor(seconds % 60)

  if (hrs > 0) {
    return `${hrs.toString().padStart(2, '0')}:${mins.toString().padStart(2, '0')}:${secs
      .toString()
      .padStart(2, '0')}`
  } else {
    return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`
  }
}

// ж јејҸеҢ–еҸ‘еёғж—¶й—?
const formatPublishTime = (timeString?: string) => {
  if (!timeString) return ''
  
  try {
    const publishTime = new Date(timeString)
    const now = new Date()
    const diffMs = now.getTime() - publishTime.getTime()
    const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24))
    
    if (diffDays === 0) {
      return 'д»ҠеӨ©еҸ‘еёғ'
    } else if (diffDays === 1) {
      return 'жҳЁеӨ©еҸ‘еёғ'
    } else if (diffDays < 7) {
      return `${diffDays}еӨ©еүҚеҸ‘еёғ`
    } else if (diffDays < 30) {
      const weeks = Math.floor(diffDays / 7)
      return `${weeks}е‘ЁеүҚеҸ‘еёғ`
    } else if (diffDays < 365) {
      const months = Math.floor(diffDays / 30)
      return `${months}дёӘжңҲеүҚеҸ‘еёғ`
    } else {
      const years = Math.floor(diffDays / 365)
      return `${years}е№ҙеүҚеҸ‘еёғ`
    }
  } catch (error) {
    return timeString
  }
}

// й”®зӣҳеҝ«жҚ·й”?
const handleKeydown = (e: KeyboardEvent) => {
  switch (e.key) {
    case 'Escape':
      e.preventDefault()
      if (isFullscreen.value) {
        toggleFullscreen()
      }
      break
    case ' ':
    case 'Spacebar':
      e.preventDefault()
      togglePlay()
      break
    case 'f':
    case 'F':
      e.preventDefault()
      toggleFullscreen()
      break
    case 'm':
    case 'M':
      e.preventDefault()
      toggleMute()
      break
    case 'd':
    case 'D':
      e.preventDefault()
      toggleDanmu()
      break
    case 'ArrowLeft':
      e.preventDefault()
      if (player.value) {
        player.value.currentTime = Math.max(0, player.value.currentTime - 5)
      }
      showControls()
      break
    case 'ArrowRight':
      e.preventDefault()
      if (player.value) {
        player.value.currentTime = Math.min(totalDuration.value, player.value.currentTime + 5)
      }
      showControls()
      break
    case 'ArrowUp':
      e.preventDefault()
      if (player.value) {
        player.value.volume = Math.min(1, player.value.volume + 0.1)
      }
      showControls()
      break
    case 'ArrowDown':
      e.preventDefault()
      if (player.value) {
        player.value.volume = Math.max(0, player.value.volume - 0.1)
      }
      showControls()
      break
  }
}
